def product(numbers):
    if numbers == []:
        return 0
    total = 1
    for n in numbers:
        total *= n
    return total

def highest_total(list1, list2):
    total1 = list_total(list1)
    total2 = list_total(list2)
    if total1 > total2:
        return list1
    else:
        return list2
    
def list_total(l):
    total = 0
    for n in l:
        total += n
    return total
        
if __name__ == "__main__":
    small = [1, 2]
    big = [2, 3, 4]
    are_equal("Both full, second bigger", highest_total(small, big), big)
    are_equal("Both full, first bigger", highest_total(big, small), big)
    are_equal("First empty", highest_total([], small), small)
    are_equal("Both empty", highest_total([], []), [])

import test

def swap(some_list, index1, index2):
    v1 = some_list[index1]
    v2 = some_list[index2]
    some_list[index1] = v2
    some_list[index2] = v1
    return some_list
        
if __name__ == "__main__":
    animals1 = ["ape", "bison", "crocodile"]
    animals2 = ["bison", "ape", "crocodile"]
    are_equal("Swap first and second", swap(animals1, 0, 1), animals2)

    highest = -9999  # weak but works for example
    for n in numbers:
        if n > highest:
            highest = n
    return highest

import test

def swap_shortest_with_longest(words):
    shortest = find_shortest_index(words)
    longest = find_longest_index(words)
    temp = words[shortest]
    words[shortest] = words[longest]
    words[longest] = temp
    return words
            
def find_longest_index(words):
    longest_word_length = -9999  # weak but works for example
    longest_word_index = -1
    for i in range(0, len(words), 1):
        if len(words[i]) > longest_word_length:
            longest_word_length = len(words[i])
            longest_word_index = i
    return longest_word_index

def find_shortest_index(words):
    shortest_word_length = 9999  # weak but works for example
    shortest_word_index = -1
    for i in range(0, len(words), 1):
        if len(words[i]) < shortest_word_length:
            shortest_word_length = len(words[i])
            shortest_word_index = i
    return shortest_word_index
    
if __name__ == "__main__":
    animals1 = ["ape", "bison", "crocodile"]
    animals2 = ["crocodile", "bison", "ape"]
    are_equal("Normal", swap_shortest_with_longest(animals1), animals2)
