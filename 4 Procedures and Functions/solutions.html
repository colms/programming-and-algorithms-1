def find_highest(x, y):
    highest = None
    if x > y:
        highest = x
    elif y > x:
        highest = y
    else:
        highest = x
    return highest
        
print(find_highest(1, 2))
print(find_highest(1, 1))
print(find_highest(2, 1))

def user_defined_range(start, stop, step):
    i = start
    l = []
    while i < stop:
        l.append(i)
        i += step
    return l

print(user_defined_range(0, 11, 2))

Function exercise 5 solution

def calculate_bmi_metric(weight_kg, height_meters):
    return weight_kg / ((height_meters) ** 2)

def find_bmi_category(bmi):
    if bmi < 18.5:
        return "underweight"
    elif bmi <= 25:
        return "normal"
    elif bmi <= 30:
        return "overweight"
    else:
        return "obese"

def display_bmi_metric(weight_kg, height_meters):
    bmi = calculate_bmi_metric(weight_kg, height_meters)
    category = find_bmi_category(bmi)
    print("Your BMI is", bmi)
    print("You are", category)

display_bmi_metric(75, 1.86)
