numbers = [4, 2, 7, 9, 4]
print("numbers is", numbers)

List exercise 2 solution

numbers = [4,2,7,9,4]
print("numbers is", numbers)
index = int(input("Enter an index: "))

List exercise 3 solution

numbers = [3, 6, 1, 4, 9]
sum = 0
stop_index = int(input("Enter an integer: ")) + 1
for index in range(0, stop_index, 1):
    sum += numbers[index]
print("sum is", sum)

List exercise 4 solution

numbers = []
sum = 0
num_inputs = 0

while num_inputs < 5:
    number = int(input("Enter an integer: "))
    numbers.append(number)
    num_inputs += 1

for index, number in enumerate(numbers):
    if number < 10:
        numbers[index] = 0
print(numbers)

List exercise 5 solution

numbers = [4,2,7,9,3]
found = False  # we haven't found the number yet
user_number = int(input("Enter an integer: "))
for number in numbers:
    if number == user_number:
        found = True
        break
if found:
    print(user_number, "is in", numbers)
else:
    print(user_number, "is not in", numbers)

List exercise 5 alternative solution

numbers = [4,2,7,9,3]
user_number = int(input("Enter an integer: "))
if user_number in numbers:
    print(user_number, "is in", numbers)
else:
    print(user_number, "is not in", numbers)

String exercise 2 solution

word = "Mississippi"
s_counter = 0
for letter in word:
    if letter == 's':
        s_counter += 1
print("The number of times the letter s occurs in Mississippi is", s_counter)

# The standard library way
print(word.count('s'))

String exercise 3 solution

a = "10001101"
b = "11001010"
differences = 0
for index in range(0,7):
    if a[index] == b[index]:
        differences += 1
print("There are", differences, "differences between", a, "and", b)

String exercise 4 solution

word = "Mississippi"
new_word = ""
for letter in word:
    if letter == 'i':
        new_word += 'a'
    else:
        new_word += letter
print("new_word is", new_word)

# The standard library way
print(word.replace('i', 'a'))

String exercise 5 solution

import random
random_word = ""
for i in range(0, 8):
    rand_int = random.randint(97, 123)
    random_word += chr(rand_int)
print(random_word)
if 'e' in random_word:
    print("e is in", random_word)
else:
    print("e is not in", random_word)
